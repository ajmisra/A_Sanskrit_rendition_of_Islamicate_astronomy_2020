% Glossary preamble

\setglossarypreamble[main]{%
This glossary lists Persian and Sanskrit technical expressions from (i) the chapter-titles of \ZijiShahJahani\ Discourse~II and \Siddhantasindhu\ Part~II in \S~\ref{chapter_title_comparision_persian_sanskrit}, (ii) the Persian text of \ZijiShahJahani\ Discourse II.6 in \S~\ref{zijshahjahan_persian_english}, and (iii) the Sanskrit text of \Siddhantasindhu\ Part II.6 in \S~\ref{siddhantasindhu_sanskrit_english}. Individual entries are grouped together under their common English translation. At the end of each entry, appropriate \textbf{chapter-numbers} and/or \textbf{passage-markers} in square brackets indicate its location in \S\S~\ref{chapter_title_comparision_persian_sanskrit}, \ref{zijshahjahan_persian_english}, or \ref{siddhantasindhu_sanskrit_english}. \Vid\ \S~\ref{glossary_appendix_format} for a more detailed description of the format of the glossary.\bigskip
}
